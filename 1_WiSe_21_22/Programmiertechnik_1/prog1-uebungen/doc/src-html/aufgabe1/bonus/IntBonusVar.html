002package aufgabe1.bonus;
003
004import java.io.BufferedReader;
005import java.io.IOException;
006import java.io.InputStreamReader;
007import java.util.NoSuchElementException;
008import java.util.Scanner;
009import java.util.concurrent.CancellationException;
010
011/**
012 * IntBonusVar zeigt den Umgang mit Variablen vom Typ int.
013 * Bonusaufgabe 1  zur Programmiertechnik 1.
014 * @author Christopher Mogler
015 * @version 21.10.2021
016 */
017public final class IntBonusVar {
018    private IntBonusVar() { }
019
020    private static final Scanner EINGABE = new Scanner(System.in);
021
022    /**
023     * Wartet auf eingabe des Users und
024     * prüft ob der User einen Integer eingegeben hat,
025     * in einer Schleifer und solange bis eine Zahl
026     * oder "exit" als Eingabe gekommen ist.
027     * Bei "exit" schmeißt die Methode eine "CancellationExecption".
028     * @return Die Eingabe des User als "long" type
029     */
030    public static long getNextNumber() {
031        String input = "";
032        while (!EINGABE.hasNextLong()) {
033            input = EINGABE.nextLine().toLowerCase().trim();
034            if (input.equals("exit")) {
035                throw new CancellationException("Abbruch vom User");
036            }
037            System.out.println("Bitte geben Sie eine Zahl ein");
038        }
039        return EINGABE.nextLong();
040    }
041
042    /**
043     * Berechnet oder Vergleicht die eingegebenen Integer,
044     * dies hängt von der jeweiligen User eingabe ab.
045     * Das Ergebnis wird über die Console zurück gegeben,
046     * Sollte Int32 für das Ergebnis nicht reichen,
047     * sowird kein Ergebnis zurückgegeben
048     * @param firstInt erster Integer Value als Long-type
049     * @param secondInt zweiter Integer Value als Long-type
050     */
051    public static void evalResult(long firstInt, Long secondInt) {
052        String input = "";
053        System.out.println(
054            "Geben Sie einen Arithmetischen- oder Vergleichsoperatoren ein: ");
055
056        boolean isRun = true;
057
058        while (isRun) {
059            input = EINGABE.next();
060            input = input.trim().toLowerCase();
061            if (input.equals("exit")) {
062                throw new CancellationException("Abbruch vom User");
063            }
064            long result = Long.MAX_VALUE;
065
066            boolean isArithmetischen = false;
067            boolean isVergleichsoperator = false;
068
069            switch (input) {
070            case "basis":
071                /* die beiden Zahlen dezimal, okatal und hexadezimal ausgeben */
072                String st = "%s ist in Oktal %o und in Hexadezimal %x%n";
073                System.out.printf(st,
074                    firstInt, firstInt, firstInt);
075                System.out.printf(st,
076                    secondInt, secondInt, secondInt);
077                break;
078            case "+":
079                result = firstInt + secondInt;
080                isArithmetischen = true;
081                break;
082            case "-":
083                result = firstInt - secondInt;
084                isArithmetischen = true;
085                break;
086            case "*":
087                result = firstInt * secondInt;
088                isArithmetischen = true;
089                break;
090            case "/":
091                result = firstInt / secondInt;
092                isArithmetischen = true;
093                break;
094            case "%":
095                result = firstInt % secondInt;
096                isArithmetischen = true;
097                break;
098            /* alle Vergleichsoperatoren ausprobieren */
099            case "==":
100                result = firstInt == secondInt ? 1 : 0;
101                isVergleichsoperator = true;
102                break;
103            case "!":
104                result = firstInt != secondInt ? 1 : 0;
105                isVergleichsoperator = true;
106                break;
107            case ">":
108                result = firstInt > secondInt ? 1 : 0;
109                isVergleichsoperator = true;
110                break;
111            case ">=":
112                result = firstInt >= secondInt ? 1 : 0;
113                isVergleichsoperator = true;
114                break;
115            case "<":
116                result = firstInt < secondInt ? 1 : 0;
117                isVergleichsoperator = true;
118                break;
119            case "<=":
120                result  = firstInt <= secondInt ? 1 : 0;
121                isVergleichsoperator = true;
122                break;
123            default:
124                break;
125            }
126            if (!isArithmetischen && !isVergleichsoperator) {
127                System.out.println("Ungültige eingabe!");
128            } else if (result > Integer.MAX_VALUE
129                || result < Integer.MIN_VALUE) {
130                System.out.print(("Das Ergebniss ist zu groß um es als 32Bit-Integer anzeigen zu lassen!"));
131            } else {
132                System.out.printf("%s %s %s ist ",
133                    firstInt, input, secondInt);
134                if (isVergleichsoperator){
135                    System.out.printf("%b %n", result);
136                }
137                else {
138                    System.out.printf("%s %n", result);                  
139                }
140                isRun = false;
141            }
142        }
143    }
144
145    /**
146     * Behandelt die Eingabe vom User.
147     */
148    public static void handleInput() {
149        int max = Integer.MAX_VALUE;
150        int min = Integer.MIN_VALUE;
151
152        System.out.println(
153            "Sie können das Program jederzeit mit 'exit' als Eingabe beenden oder mit 'Strg' + 'D' auf Unix und 'Strg' + 'F' auf Windows!");
154        /* Eingabeaufforderung ausgeben */
155        System.out.printf("Zwei ganze Zahlen zwischen %s und %s eingeben%n",
156            max, min);
157
158        /* zwei ganze Zahlen einlesen */
159        long firstInt = getNextNumber();
160        long secondInt = getNextNumber();
161
162        evalResult(firstInt, secondInt);
163    }
164
165    /**
166     * main ist der Startpunkt des Programms.
167     * @param args wird nicht verwendet.
168     */
169    @SuppressWarnings("magicnumber")
170    public static void main(String[] args) {
171        BufferedReader buf = new BufferedReader (new InputStreamReader(System.in));
172        try {
173            String inp = "";
174            do {
175                handleInput();
176                System.out.printf(
177                    "%nWenn Sie erneut eine Eingabe tätigen wollen, drücken Sie die 'y'-Taste und bestätigen Sie mit der 'Enter'-Taste."
178                );
179                inp = buf.readLine();
180                inp = inp.trim().toLowerCase();
181            } while (inp.equals("y"));
182
183        } catch (CancellationException | NoSuchElementException e) {
184            System.out.println("Programm wurde vom User beendet.");
185        } catch (IOException e) {
186            System.out.println("Programm wurde unerwartet beendet.");
187        }
188    }
189}
