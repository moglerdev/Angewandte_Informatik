003package aufgabe1;
004
005import java.util.Scanner;
006
007/**
008 * IntVar zeigt den Umgang mit Variablen vom Typ int.
009 * &Uuml;bungsaufgabe 1 zur Programmiertechnik 1.
010 * @author Christopher Mogler
011 * @version 19.10.2021
012 */
013public final class IntVar {
014    private IntVar() { }
015
016    private static final Scanner EINGABE = new Scanner(System.in);
017
018    /**
019     * main ist der Startpunkt des Programms.
020     * @param args wird nicht verwendet.
021     */
022    @SuppressWarnings("magicnumber")
023    public static void main(String[] args) {
024        int min = 0x80000000;
025        int max = 0x7fffffff;
026
027        /* Eingabeaufforderung ausgeben */
028        System.out.printf("Zwei ganze Zahlen zwischen %s und %s eingeben%n",
029            min, max);
030
031        /* zwei ganze Zahlen einlesen */
032        int firstInt = EINGABE.nextInt();
033        int secondInt = EINGABE.nextInt();
034
035        /* die beiden Zahlen dezimal, okatal und hexadezimal ausgeben */
036        System.out.printf("%s ist oktal %o und hexadezimal %x%n",
037            firstInt, firstInt, firstInt);
038        System.out.printf("%s ist oktal %o und hexadezimal %x%n",
039            secondInt, secondInt, secondInt);
040
041        /* alle zweistelligen arithmetischen Operatoren ausprobieren */
042        System.out.printf("%s + %s ist %s%n",
043            firstInt, secondInt, firstInt + secondInt);
044        System.out.printf("%s - %s ist %s%n",
045            firstInt, secondInt, firstInt - secondInt);
046        System.out.printf("%s * %s ist %s%n",
047            firstInt, secondInt, firstInt * secondInt);
048        System.out.printf("%s / %s ist %s%n",
049            firstInt, secondInt, firstInt / secondInt);
050        System.out.printf("%s %% %s ist %s%n",
051            firstInt, secondInt, firstInt % secondInt);
052
053        /* alle Vergleichsoperatoren ausprobieren */
054        System.out.printf("%s == %s ist %b%n",
055            firstInt, secondInt, firstInt == secondInt);
056        System.out.printf("%s != %s ist %b%n",
057            firstInt, secondInt, firstInt != secondInt);
058        System.out.printf("%s < %s ist %b%n",
059            firstInt, secondInt, firstInt < secondInt);
060        System.out.printf("%s <= %s ist %b%n",
061            firstInt, secondInt, firstInt <= secondInt);
062        System.out.printf("%s > %s ist %b%n",
063            firstInt, secondInt, firstInt > secondInt);
064        System.out.printf("%s >= %s ist %b%n",
065            firstInt, secondInt, firstInt >= secondInt);
066    }
067}
