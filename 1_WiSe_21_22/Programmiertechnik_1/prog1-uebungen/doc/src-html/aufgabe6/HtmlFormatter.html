001package aufgabe6;
002
003import java.util.Scanner;
004import java.io.PrintWriter;
005import java.io.File;
006import java.io.IOException;
007
008/**
009 * Ist dafür zuständig das Ergebnis in eine HTML Datei zuschreiben.
010 * @author Christopher Mogler
011 * @version 17.01.2022
012 */
013public final class HtmlFormatter {
014    private HtmlFormatter() { }
015    private static final String NOT_PASSED_CSS_CLASS
016        = "not_passed";
017    private static final String TEMPLATE_FILE
018        = "aufgabe6/assets/template.min.html";
019
020    /**
021     * Schreibt den formatierten HTML String in die
022     *  referenzierte <code>output</code> variable.
023     * @param output PrintWriter wird benötigt für den output
024     * @param name Der Name des Studenten
025     * @param ratingList die Leistungen des Studenten
026     * @throws IOException wenn Datei nicht gefunden wurde.
027     */
028    public static void write(
029        PrintWriter output,
030        String[] name,
031        RatingList ratingList)
032        throws IOException {
033
034        String htmlRatingList = "";
035        String type = "";
036        String grade = "";
037
038        File templateFile = new File(TEMPLATE_FILE);
039
040        for (Rating rating : ratingList) {
041            grade = rating.getGradeInWords();
042
043            if (rating.isGraded()) {
044                type = "S";
045                grade += " (" + rating.getGrade() + ")";
046            } else {
047                type = "L";
048            }
049            htmlRatingList += "<tr"
050                + " class='"
051                + (!rating.isPassed() ? NOT_PASSED_CSS_CLASS : "")
052                + "'>"
053                + "<td>" + rating.getSubject() + "</td>"
054                + "<td>" + type + "</td>"
055                + "<td>" + grade + "</td></tr>";
056        }
057
058        Scanner r = new Scanner(templateFile);
059        String data = r.nextLine()
060            .replace("{{ name }}", String.join(" ", name))
061            .replace("{{ ratingList }}", htmlRatingList);
062
063        output.write(data);
064        r.close();
065    }
066}
