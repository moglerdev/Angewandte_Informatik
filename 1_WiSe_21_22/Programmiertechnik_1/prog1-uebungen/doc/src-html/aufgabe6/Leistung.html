002package aufgabe6;
003
004/**
005 * Leistung kapselt ein Fach mit Note als Entit&auml;t.
006 * @author H.Drachenfels
007 * @version 18.12.2018
008 */
009public abstract class Leistung {
010    private final String fach;
011
012    /**
013     * initialisiert die Fachnote mit einem Namen.
014     * <p>
015     * Wirft eine Ausnahme <code>IllegalArgumentException</code>, wenn das Fach
016     * laut {@link Faecher#istZulaessig} nicht zul&auml;ssig ist.</p>
017     * @param f der Fachname
018     */
019    protected Leistung(String f) {
020        if (!Faecher.istZulaessig(f)) {
021            throw new IllegalArgumentException("unzulaessiges Fach " + f);
022        }
023
024        this.fach = f;
025    }
026
027    /**
028     * liefert den Namen des Fachs.
029     * @return den Fachnamen
030     */
031    public final String getFach() {
032        return this.fach;
033    }
034
035    /**
036     * liefert die Note des Fachs in numerischer Schreibweise.
037     * <p>
038     * Wenn {@link #istBenotet} <code>true</code> zur&uuml;ckgibt,
039     * muss die Methode in der Unterklasse so &uuml;berschrieben werden,
040     * dass sie die Note in numerischer Schreibweise liefert,
041     * also Strings "1,0", "1,3" usw.
042     * @return leerer String &quot;&quot;
043     */
044    public String getNote() {
045        assert !istBenotet() : this.getClass().getName()
046                               + " muss getNote() ueberschreiben";
047        return "";
048    }
049
050    /**
051     * liefert die Note des Fachs in Worten.
052     * <p>
053     * Wenn {@link #istBenotet} <code>true</code> zur&uuml;ckgibt,
054     * muss die Methode in der Unterklasse &uuml;berschrieben werden.
055     * Der geliefert String h&auml;ngt dann davon ab,
056     * welches Notensystem die Unterklasse verwendet.
057     * @return "bestanden", wenn {@link #istBestanden} true liefert,
058     *         sonst "nicht bestanden".
059     */
060    public String getNoteInWorten() {
061        assert !istBenotet() : this.getClass().getName()
062                               + " muss getNoteInWorten() ueberschreiben";
063        if (istBestanden()) {
064            return "bestanden";
065        }
066
067        return "nicht bestanden";
068    }
069
070    /**
071     * fragt ab, ob das Fach bestanden ist.
072     * Wenn {@link #istBenotet} <code>true</code> zur&uuml;ckgibt,
073     * h&auml;ngt es vom Notensystem der Unterklasse ab,
074     * welche Noten als bestanden und welche als nicht bestanden gelten.
075     * @return <code>true</code>, wenn bestanden, sonst <code>false</code>
076     */
077    public abstract boolean istBestanden();
078
079    /**
080     * fragt ab, ob das Fach benotet oder unbenotet ist.
081     * <p>
082     * Darf nur <code>true</code> liefern, wenn die Methoden
083     * {@link #getNote} und {@link #getNoteInWorten}
084     * so &uuml;berschreiben sind, dass sie eine Note liefern.</p>
085     * @return <code>true</code>, wenn benotet, sonst <code>false</code>
086     */
087    public abstract boolean istBenotet();
088}
