004/**
005 * Fächer.
006 * @author Christopher Mogler
007 * @version 17.01.2022
008 */
009public final class Subjects {
010    private Subjects() { }
011
012    private static final String[] SUBJECTS = {
013        "Programmiertechnik 1",
014        "Softwaremodelierung",
015        "Mathematik 1",
016        "Digitaltechnik"
017    };
018
019    /**
020     * Prüft ob das angegebene Fach an der HTWG AIN 1 existiert.
021     * @param subject Fachname was geprüft werden soll
022     * @return <code>true</code> wenn das Fach
023     * gültig ist, <code>false</code> wenn nicht.
024     */
025    public static boolean isValid(String subject) {
026        for (String s : SUBJECTS) {
027            if (subject.equals(s)) {
028                return true;
029            }
030        }
031        return false;
032    }
033
034    /**
035     * Gibt die Anzahl an Buchstaben zurück von
036     * dem Fach mit den meisten Buchstaben.
037     * @return länge des größten Fachs, nach Buchstaben
038     */
039    public static int getMaxLengthOfSubject() {
040        int result = 0;
041        for (String s : SUBJECTS) {
042            if (result < s.length()) {
043                result = s.length();
044            }
045        }
046        return result;
047    }
048}
