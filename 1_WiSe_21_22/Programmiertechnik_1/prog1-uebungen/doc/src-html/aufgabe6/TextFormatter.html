001package aufgabe6;
002
003import java.io.IOException;
004import java.io.PrintWriter;
005
006/**
007 * Ist dafür zuständig das Ergebnis in eine TXT Datei zuschreiben.
008 * @author Christopher Mogler
009 * @version 17.01.2022
010 */
011public final class TextFormatter {
012    private TextFormatter() { }
013
014    /**
015     * Schreibt den formatierten TXT String in die
016     * referenzierte <code>output</code> variable.
017     * @param output PrintWriter wird benötigt für den output
018     * @param name Der Name des Studenten
019     * @param ratingList die Leistungen des Studenten
020     * @throws IOException sollte die Datei nicht erstellt werden können
021     */
022    public static void write(
023        PrintWriter output,
024        String[] name,
025        RatingList ratingList)
026        throws IOException {
027        String placeholder = new String(
028            new char[Subjects.getMaxLengthOfSubject()])
029                .replace('\0', ' ');
030
031        String tableRatingList = "Notenspiegel für " + String.join(" ", name)
032            + "\r\nFach" + placeholder + "Art\t\tNote";
033        tableRatingList += "\r\n" + new String(
034            new char[tableRatingList.length()])
035                .replace('\0', '-') + "\r\n" + "\r\n";
036
037        String type = "";
038        String grade = "";
039
040        final int bufferLength = 4;
041
042        for (Rating rating : ratingList) {
043            placeholder = new String(
044                new char[bufferLength + Subjects.getMaxLengthOfSubject()
045                    - rating.getSubject().length()]).replace('\0', ' ');
046            grade = rating.getGradeInWords();
047
048
049
050            if (rating.isGraded()) {
051                type = "L";
052                grade += " (" + rating.getGrade() + ")";
053            } else {
054                type = "S";
055            }
056            tableRatingList += rating.getSubject()
057                + placeholder + type
058                + "\t\t" + grade + "\r\n";
059        }
060
061        output.write(tableRatingList);
062    }
063}
