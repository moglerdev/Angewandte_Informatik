002package aufgabe6;
003
004import java.util.NoSuchElementException;
005
006import java.util.Iterator;
007
008/**
009 * LeistungsListe kapselt eine Liste von Leistungen.
010 * @author Christopher Mogler
011 * @version 13.01.2022
012 */
013public final class RatingList implements Iterable<Rating> {
014    private Element head = null;
015
016    /**
017     * Konstruktor für ratingList.
018     * @param ratings Anfangsbestand der Liste (wird sortiert).
019     */
020    public RatingList(Rating... ratings) {
021        for (Rating rating : ratings) {
022            this.insert(rating);
023        }
024    }
025
026    /**
027     * fügt die Note der Liste hinzu (UNIQUE!).
028     * @param rating neue Note
029     * @return index der Note in der Liste
030     */
031    public Rating insert(Rating rating) {
032        if (rating == null) {
033            throw new IllegalArgumentException();
034        }
035        Rating deletetElement = null;
036        Element newEle = new Element(rating);
037
038        if (this.head == null) {
039            this.head = newEle;
040        } else {
041            Element curr = this.head;
042            while (true) {
043                Rating test = curr.value;
044                if (rating.isGraded()
045                    && rating.getSubject().equals(test.getSubject())) {
046                    deletetElement = curr.value;
047                    curr.value = rating;
048                    break;
049                } else if (curr.next == null) {
050                    curr.next = newEle;
051                    curr = null;
052                    break;
053                }
054                curr = curr.next;
055            }
056        }
057        return deletetElement;
058    }
059
060    /**
061     * ganzes Array hinzufügen (wird sortiert!).
062     * @param ratings Array von Noten
063     */
064    public void insert(Rating... ratings) {
065        if (ratings != null && ratings.length > 0) {
066            for (Rating rating : ratings) {
067                this.insert(rating);
068            }
069        }
070    }
071
072    /**
073     * gibt Note über index zurück.
074     * @param index des Objekts
075     * @return Objekt über index in der Liste
076     */
077    public Rating get(int index) {
078        if (index > -1) {
079            Element result = this.head;
080            for (int i = 1; i < index; ++i) {
081                if (result.next == null) {
082                    throw new IndexOutOfBoundsException();
083                }
084                result = result.next;
085            }
086
087            return result.value;
088        }
089
090        throw new IndexOutOfBoundsException();
091    }
092
093    /**
094     * gibt die aktuelle Länge der Liste zurück.
095     * @return länge als int
096     */
097    public int length() {
098        Element ele = this.head;
099        int result = 0;
100        while (ele != null) {
101            ++result;
102            ele = ele.next;
103        }
104        return result;
105    }
106
107    /**
108     * um diese Klasse Iterable zu machen (bspw. foreach)
109     * @return new NIterator
110     */
111    @Override()
112    public Iterator<Rating> iterator() {
113        return new NIterator(this);
114    }
115
116    /** ratingItator Klasse für Iterable (bspw. foreach). */
117    public static final class NIterator implements Iterator<Rating> {
118        private Element current = null;
119
120        /**
121         * Konstruktor mit zur iterierender Liste.
122         * @param list aktuelle Liste
123         */
124        NIterator(RatingList list) {
125            this.current = list.head;
126        }
127
128        /**
129         * ob es ein nächstes Objekt in der Liste gibt.
130         * @return true == ja
131         */
132        public boolean hasNext() {
133            return this.current != null;
134        }
135
136        /**
137         * nächstes Item von der Liste.
138         * @return nänstes Item von der Liste
139         */
140        public Rating next() {
141            if (this.hasNext()) {
142                Element e = this.current;
143                this.current = this.current.getNext();
144                return e.getValue();
145            }
146
147            throw new NoSuchElementException();
148        }
149
150        /**
151         * Löschen (UnsupportedOperationException!).
152         * throws UnsupportedOperationException
153         */
154        public void remove() {
155            throw new UnsupportedOperationException();
156        }
157    }
158
159
160    /** Element Klasse für die Liste. */
161    private static final class Element {
162        /** nächstes Item der Liste. */
163        private Element next = null;
164        /** Value (rating) von dem Item in der Liste. */
165        private Rating value = null;
166
167        /**
168         * Konstruktor für das Element.
169         * @param value aktueller Wert
170         */
171        Element(Rating value) {
172            this.value = value;
173        }
174
175        /**
176         * gibt das Nächste Item der Liste zurück.
177         * @return Element nächstes Item
178         */
179        public Element getNext() {
180            return next;
181        }
182
183        /**
184         * gibt aktuelle Value zurück.
185         * @return value als rating
186         */
187        public Rating getValue() {
188            return value;
189        }
190    }
191}
