002package aufgabe6;
003import java.io.BufferedReader;
004import java.io.InputStreamReader;
005import java.io.PrintWriter;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008
009/**
010 * Die Main-Klasse HtmlNotenspiegel liest die Namen von F&auml;chern
011 * mit den zugeh&ouml;rigen Noten in eine verkettete Liste ein und
012 * gibt dann einen Notenspiegel im HTML-Format aus.
013 * Aufruf mit dem Namen der Studentin / des Studenten als Argument(e)
014 * @author H.Drachenfels
015 * @version 28.7.2021
016 */
017public final class HtmlNotenspiegel {
018    private HtmlNotenspiegel() { }
019
020    /**
021     * liest die Namen von F&auml;chern mit den zugeh&ouml;rigen Beurteilungen
022     * ein und schreibt einen Notenspiegel als Datei im HTML-Format.
023     * <p>
024     * Jede Eingabezeile muss einen Fachnamen und eine Beurteilung enthalten.
025     * Fachnamen m&uuml;ssen laut
026     * {@link aufgabe6.Faecher#istZulaessig} erlaubt sein.
027     * Bei unbenoteten F&auml;chern muss die Beurteilung
028     * <code>BE</code> f&uuml;r bestanden oder
029     * <code>NB</code> f&uuml;r nicht bestanden lauten.
030     * Bei benoteten F&auml;chern muss die Note laut
031     * {@link aufgabe5.Note#valueOf(java.lang.String)} erlaubt sein.
032     * Der Dateinname des Notenspiegels wird aus den Argumenten gebildet.</p>
033     * @param args Vorname(n) Nachname (mindestens zwei Argumente)
034     * @throws IOException bei Einleseproblemen
035     */
036    public static void main(String[] args) throws IOException {
037        if (args.length < 2) {
038            System.err.println(
039                "Aufruf: java HtmlNotenspiegel Vorname(n) Nachname");
040            System.exit(1);
041        }
042
043        LeistungsListe leistungen = new LeistungsListe();
044        einlesen(leistungen);
045        ausgeben(args, leistungen);
046    }
047
048    private static void einlesen(LeistungsListe leistungen)
049        throws IOException {
050
051        System.err.println("Faecher mit Noten eingeben (Ende mit Strg-D):");
052        BufferedReader eingabe
053            = new BufferedReader(new InputStreamReader(System.in));
054
055        String zeile;
056        while ((zeile = eingabe.readLine()) != null) {
057            zeile = zeile.trim();
058            if (zeile.length() == 0) {
059                continue; // Leerzeile
060            }
061
062            try {
063                String[] woerter = zeile.split("\\s+");
064                if (woerter.length < 2) {
065                    throw new IllegalArgumentException(
066                        "unvollstaendige Zeile " + zeile);
067                }
068
069                String note = woerter[woerter.length - 1];
070                String fach = zeile.substring(0, zeile.lastIndexOf(note));
071
072                Leistung fachnote;
073                if (note.equals("BE")) {
074                    fachnote = new UnbenoteteLeistung(fach.trim(), true);
075                } else if (note.equals("NB")) {
076                    fachnote = new UnbenoteteLeistung(fach.trim(), false);
077                } else {
078                    fachnote = new BenoteteLeistung(fach.trim(), note);
079                }
080
081                leistungen.insert(fachnote);
082            } catch (IllegalArgumentException x) {
083                System.err.printf("Falscheingabe ignoriert: %s%n",
084                                  x.getMessage());
085                continue;
086            }
087        }
088    }
089
090    private static void ausgeben(String[] name, LeistungsListe leistungen)
091        throws IOException {
092
093        StringBuilder sb = new StringBuilder();
094        for (String s : name) {
095            sb.append(s);
096        }
097
098        try {
099            PrintWriter ausgabe
100                = new PrintWriter(sb.toString() + ".html", "UTF-8");
101            HtmlFormat.ausgeben(ausgabe, name, leistungen);
102            ausgabe.close();
103        } catch (FileNotFoundException x) {
104            System.err.println(x.getMessage());
105            System.exit(1);
106        }
107    }
108}
