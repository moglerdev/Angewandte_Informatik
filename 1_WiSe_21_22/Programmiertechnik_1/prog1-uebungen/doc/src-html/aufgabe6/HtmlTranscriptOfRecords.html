002package aufgabe6;
003import java.io.PrintWriter;
004import java.io.BufferedReader;
005import java.io.FileNotFoundException;
006import java.io.InputStreamReader;
007import java.io.IOException;
008
009/**
010 * Die Main-Klasse HtmlNotenspiegel liest die Namen von F&auml;chern
011 * mit den zugeh&ouml;rigen Noten in eine verkettete Liste ein und
012 * gibt dann einen Notenspiegel im HTML-Format aus.
013 * Aufruf mit dem Namen der Studentin / des Studenten als Argument(e)
014 * @author Christopher Mogler
015 * @version 17.01.2022
016 */
017public final class HtmlTranscriptOfRecords {
018    private HtmlTranscriptOfRecords() { }
019
020
021    private static final String OUTPUT_PATH
022        = "aufgabe6/output/";
023
024    /**
025     * liest die Namen von F&auml;chern mit den zugeh&ouml;rigen Beurteilungen
026     * ein und schreibt einen Notenspiegel als Datei im HTML-Format.
027     * <p>
028     * Jede Eingabezeile muss einen Fachnamen und eine Beurteilung enthalten.
029     * Fachnamen m&uuml;ssen laut
030     * {@link aufgabe6.Subjects#isValid} erlaubt sein.
031     * Bei unbenoteten F&auml;chern muss die Beurteilung
032     * <code>BE</code> f&uuml;r bestanden oder
033     * <code>NB</code> f&uuml;r nicht bestanden lauten.
034     * Bei benoteten F&auml;chern muss die Note laut
035     * {@link aufgabe5.Grade#valueOf(java.lang.String, java.lang.String)}
036     * erlaubt sein.
037     * Der Dateinname des Notenspiegels wird aus den Argumenten gebildet.</p>
038     * @param args Vorname(n) Nachname (mindestens zwei Argumente)
039     * @throws IOException bei Einleseproblemen
040     */
041    public static void main(String[] args) throws IOException {
042        if (args.length < 2) {
043            System.err.println(
044                "Aufruf: java HtmlNotenspiegel Vorname(n) Nachname");
045            System.exit(1);
046        }
047
048        RatingList ratingList = new RatingList();
049        read(ratingList);
050        write(args, ratingList);
051    }
052
053    /**
054     * List die Leistungen über die Befehlszeile ein
055     * und evaluiert sie und schreibt diese in die ref. Liste.
056     * @param ratingList die referenzierte
057     *  Leistungs-Liste die befüllt werden soll.
058     * @throws IOException BufferedReader
059     */
060    private static void read(RatingList ratingList)
061        throws IOException {
062
063        System.err.println("Faecher mit Noten eingeben (Ende mit Strg-D):");
064        BufferedReader input
065            = new BufferedReader(new InputStreamReader(System.in));
066
067        String line;
068        while ((line = input.readLine()) != null) {
069            line = line.trim();
070            if (line.length() == 0) {
071                continue; // Leerzeile
072            }
073
074            try {
075                String[] words = line.split("\\s+");
076                if (words.length < 2) {
077                    throw new IllegalArgumentException(
078                        "unvollstaendige Zeile " + line);
079                }
080
081                String grade = words[words.length - 1];
082                String subject = line.substring(0, line.lastIndexOf(grade));
083
084                Rating rating;
085                if (grade.equals("BE")) {
086                    rating = new UngradedRating(subject.trim(), true);
087                } else if (grade.equals("NB")) {
088                    rating = new UngradedRating(subject.trim(), false);
089                } else {
090                    rating = new GradedRating(subject.trim(), grade.trim());
091                }
092
093                Rating r = ratingList.insert(rating);
094                if (r != null) {
095                    System.out.printf(
096                        "Warnung: Leistung %s wurde überschrieben."
097                        + "(Doppelter Eintrag)%n",
098                        r.getSubject());
099                }
100            } catch (IllegalArgumentException x) {
101                System.err.printf("Falscheingabe ignoriert: %s%n",
102                                  x.getMessage());
103                continue;
104            }
105        }
106    }
107
108    /**
109     * Schreibt die evaluierte Liste in eine HTML Datei.
110     * @param name Vor- und Nachname als Array
111     * @param ratingList Liste der Leistungen
112     * @throws IOException Sollte eine Datei nicht gefunden werden.
113     */
114    private static void write(String[] name, RatingList ratingList)
115        throws IOException {
116        String outputFilename = OUTPUT_PATH + String.join("", name);
117
118        try (
119            PrintWriter htmlOutput
120                = new PrintWriter(outputFilename + ".html", "UTF-8");
121            PrintWriter txtOutput
122                = new PrintWriter(outputFilename + ".txt", "UTF-8");
123        ) {
124            HtmlFormatter.write(htmlOutput, name, ratingList);
125            TextFormatter.write(txtOutput, name, ratingList);
126        } catch (FileNotFoundException e) {
127            System.err.printf("ERROR: File not found: %s%n", e.getMessage());
128        }
129    }
130}
