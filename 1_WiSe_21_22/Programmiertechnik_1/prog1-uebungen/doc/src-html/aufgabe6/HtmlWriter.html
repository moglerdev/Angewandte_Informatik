001package aufgabe6;
002
003import java.io.PrintWriter;
004import java.util.Scanner;
005import java.io.File;
006import java.io.FileNotFoundException;
007
008public final class HtmlWriter {
009    private static final String TEMPLATE_FILE = "assets/template.min.html";
010    private static final String NOT_PASSED_CSS_CLASS = "NOT_PASSED";
011
012    public static void write(PrintWriter output, String[] name, RatingList ratingList) {
013        try {
014            String htmlRatingList = "";
015
016            File templateFile = new File(TEMPLATE_FILE);
017            
018            for (Rating rating : ratingList) {
019                htmlRatingList += "<tr" + 
020                    "class='" + (!rating.isPassed() ? NOT_PASSED_CSS_CLASS : "") + "'>" +
021                    "<td>" + rating.getSubject() + "</td>" +
022                    "<td>" + (rating instanceof UngradedRating ? "S" : "L") + "</td>" +
023                    "<td>" + rating.getGradeInWords() + "</td></tr>";
024            }
025
026            Scanner r = new Scanner(templateFile);
027            String data = r.nextLine().replace("{{ ratingList }}", htmlRatingList);
028
029            output.write(data);
030
031            r.close();
032        } catch (FileNotFoundException e) {
033            System.out.println("An error occurred.");
034            e.printStackTrace();
035        }
036    }
037}
