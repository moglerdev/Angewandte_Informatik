002package aufgabe2.bonus;
003
004import java.util.ArrayList;
005import java.util.HashMap;
006import java.util.Map;
007import java.util.Scanner;
008
009/**
010 * HistogrammBonus liest ganze Zahlen zwischen 1 und 6 ein und
011 * gibt deren H&auml;ufigkeitsverteilung als Histogramm aus.
012 * @author Christopher Mogler
013 * @version 02.11.2021
014 */
015@SuppressWarnings("magicnumber")
016public final class HistogrammBonus {
017    private HistogrammBonus() { }
018
019    private static final Scanner EINGABE = new Scanner(System.in);
020
021    private static final int MAX_LINE_LENGTH = 50;
022
023    /**
024     * Zeichnet das Histogramm Horizontal.
025     * @param numbersMap Nummer Map, die ausgegeben werden soll.
026     * @param n Die Sprünger einzelner Zahlen
027     * @param maxSize die größte Länge einer Liste der numbersMap
028     * @param minSize die kleinste Länge einer Liste der numbersMap
029     * @param isCutting soll die Ausgabe zugeschnietten werden?
030     */
031    public static void drawHorizontal(Map<Integer, Integer[]>  numbersMap, int n, int maxSize, int minSize, boolean isCutting) {
032        for (int o = 1; o < 7; ++o) {
033            Integer[] oA = numbersMap.get(o);
034
035            if (oA != null) {
036                for (int num = (isCutting ? minSize - 1 : 0); num < oA.length; num += n) {
037                    System.out.printf("%d", oA[num]);
038                }
039
040                System.out.printf(" (%d)%n", oA.length);
041            }
042        }
043    }
044
045    /**
046     * Zeichnet das Histogramm Vertikal.
047     * @param numbersMap Nummer Map, die ausgegeben werden soll.
048     * @param n Die Sprünger einzelner Zahlen
049     * @param maxSize die größte Länge einer Liste der numbersMap
050     * @param minSize die kleinste Länge einer Liste der numbersMap
051     * @param isCutting ob der Wert gecuttet werden soll
052     */
053    public static void drawVertical(Map<Integer, Integer[]>  numbersMap, int n, int maxSize, int minSize, boolean isCutting) {
054
055        for (int i = (isCutting ? minSize - 1 : 0); i < maxSize; i = i + n) {
056
057            for (Map.Entry<Integer, Integer[]> entry : numbersMap.entrySet()) {
058                Integer[] numbers = entry.getValue();
059
060                if (numbers.length > i) {
061                    System.out.printf("%d \t", numbers[i]);
062                } else {
063                    System.out.printf("\t");
064                }
065            }
066
067            System.out.printf("(%d)%n", i + 1);
068        }
069    }
070
071    /**
072     * Zeichnet das Histogramm.
073     * @param numbers für die Ausgabe
074     * @param isVertical Soll es Vertikal gezeichnet werden, anstatt Horizontal
075     * @param isCutting Soll es gekürzt werden
076     */
077    public static void drawHistogramm(ArrayList<Integer> numbers, boolean isVertical, boolean isCutting) {
078        Map<Integer, Integer[]> outMap = new HashMap<Integer, Integer[]>();
079        ArrayList<Integer> outNrList = new ArrayList<Integer>();
080
081        Integer[] outTmpNrList;
082        int maxSize = 0;
083        int minSize = numbers.size();
084
085        for (int i = 1; i < 7; ++i) {
086            outNrList.clear();
087
088            for (int k = 0; k < numbers.size(); ++k) {
089                int nr = numbers.get(k);
090                if (nr == i) {
091                    outNrList.add(nr);
092                }
093            }
094
095            if (maxSize < outNrList.size()) {
096                maxSize = outNrList.size();
097            }
098            if (minSize > outNrList.size()) {
099                minSize = outNrList.size();
100            }
101
102            if (outNrList.size() > 0) {
103                outTmpNrList = new Integer[outNrList.size()];
104                outNrList.toArray(outTmpNrList);
105                outMap.put(i, outTmpNrList);
106            }
107        }
108
109        int n = maxSize / MAX_LINE_LENGTH + 1;
110
111        System.out.printf("Histrogramm: %d %n", maxSize);
112
113        if (isVertical) {
114            drawVertical(outMap, n, maxSize, minSize, isCutting);
115        } else {
116            drawHorizontal(outMap, n, maxSize, minSize, isCutting);
117        }
118    }
119
120    /**
121     * main ist der Startpunkt des Programms.
122     * @param args wird nicht verwendet.
123     */
124    @SuppressWarnings("magicnumber")
125    public static void main(String[] args) {
126
127        boolean isCutting = false;
128        boolean isVertical = false;
129
130        for (String arg : args) {
131            if (arg.equals("-CUT")) { 
132                isCutting = true;
133            }
134            if (arg.equals("-VERTICAL")) {
135                isVertical = true;
136            }
137        }
138
139        /* (1) hier ein Feld von Zaehlern definieren */
140        ArrayList<Integer> numbers = new ArrayList<Integer>();
141
142        //---------------------------------------------------- Zahlen einlesen
143        System.out.println("Ganze Zahlen zwischen 1 und 6 eingeben "
144                    + "(Ende mit Strg-D/Strg-Z):");
145
146        while (EINGABE.hasNextInt()) {
147            int number = EINGABE.nextInt();
148
149            /* (2) hier Anweisungen fuer das
150                         Pruefen und Zaehlen der Eingabe schreiben */
151
152            if (number < 1 || number > 7) {
153                System.out.printf("Falsche Eingabe wird ingoriert: %s%n", number);
154            } else {
155                numbers.add(number);
156            }
157        }
158
159        //------------------------------------------------ Histogramm ausgeben
160
161        /* (3) hier Anweisungen fuer die Histogrammausgabe schreiben */
162        drawHistogramm(numbers, isVertical, isCutting);
163    }
164}
