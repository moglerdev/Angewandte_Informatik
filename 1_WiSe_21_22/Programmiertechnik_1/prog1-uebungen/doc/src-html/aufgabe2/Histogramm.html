002package aufgabe2;
003
004import java.util.ArrayList;
005import java.util.Scanner;
006
007/**
008 * Histogramm liest ganze Zahlen zwischen 1 und 6 ein und
009 * gibt deren H&auml;ufigkeitsverteilung als Histogramm aus.
010 * @author Christopher Mogler
011 * @version 02.11.2021
012 */
013public final class Histogramm {
014    private Histogramm() { }
015
016    private static final Scanner EINGABE = new Scanner(System.in);
017
018    /**
019     * main ist der Startpunkt des Programms.
020     * @param args wird nicht verwendet.
021     */
022    @SuppressWarnings("magicnumber")
023    public static void main(String[] args) {
024
025        /* (1) hier ein Feld von Zaehlern definieren */
026        ArrayList<Integer> numbers = new ArrayList<Integer>();
027
028        //---------------------------------------------------- Zahlen einlesen
029        System.out.println("Ganze Zahlen zwischen 1 und 6 eingeben "
030                    + "(Ende mit Strg-D/Strg-Z):");
031
032        while (EINGABE.hasNextInt()) {
033            int number = EINGABE.nextInt();
034
035            /* (2) hier Anweisungen fuer das
036                         Pruefen und Zaehlen der Eingabe schreiben */
037
038            if (number < 1 || number > 7) {
039                System.out.printf("Falsche Eingabe wird ingoriert: %s%n", number);
040            } else {
041                numbers.add(number);
042            }
043        }
044
045        //------------------------------------------------ Histogramm ausgeben
046
047        /* (3) hier Anweisungen fuer die Histogrammausgabe schreiben */
048        System.out.println("Histrogramm:");
049        for (int i = 1; i < 7; ++i) {
050            int count = 0;
051
052            for (int k = 0; k < numbers.size(); ++k) {
053                int n = numbers.get(k);
054                if (n == i) {
055                    ++count;
056                    if (count % 5 == 0) {
057                        System.out.print("$");
058                    } else {
059                        System.out.printf("%s", n);
060                    }
061                }
062            }
063            if (count > 0) {
064                System.out.printf(" (%s)%n", count);
065            }
066        }
067    }
068}
