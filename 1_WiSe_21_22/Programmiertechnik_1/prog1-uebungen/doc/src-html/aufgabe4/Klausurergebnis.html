001// Klausurergebnis.java
002package aufgabe4;
003import aufgabe4.switzerland.Grades;
004import java.util.Locale;
005import java.util.Scanner;
006
007/**
008 * Klausurergebnis erstellt eine Notenstatistik.
009 * <p>
010 * Das Programm liest Noten als Strings ein und bestimmt die beste und
011 * die schlechteste Note, den Durchschnitt der Bestandenen sowie
012 * die Durchfallquote in Prozent.
013 * Das Programm ber&uuml;cksichtigt dabei nur die laut Noten.istZulaessig
014 * erlaubten Noten. Andere Noten werden unter Ausgabe einer Warnung ignoriert.
015 * Welche Noten besser und schlechter sind, welche als bestanden oder
016 * durchgefallen gelten und wie die String-Darstellung der Noten aussieht,
017 * wird mit Methoden der Klasse Noten bestimmt.
018 * </p>
019 * Das Programm gibt als Klausurergebnis folgende Werte aus:
020 * <ul>
021 * <li>die Anzahl der ber&uuml;cksichtigten Noten</li>
022 * <li>die Anzahl der Bestandenen</li>
023 * <li>die beste Note</li>
024 * <li>die schlechteste Note</li>
025 * <li>den Durchschnitt der Bestandenen</li>
026 * <li>die Durchfallquote</li>
027 * </ul>
028 *
029 * @author Christopher Mogler
030 * @version 02.12.2021
031 */
032public final class Klausurergebnis {
033    private Klausurergebnis() { }
034
035    /** Beste Note. */
036    private static double bestGrade = Grades.SCHLECHTESTE;
037    /** Schlechteste Note. */
038    private static double worstGrade = Grades.BESTE;
039
040    /** Anzahl aller korekten Noten. */
041    private static int countGrades = 0;
042    /** Anzahl aller bestandenen Noten. */
043    private static int countPassed = 0;
044
045    /** Die Summe aller Noten, für den Durchschnitt. */
046    private static double sumOfPassedGrades = 0.0;
047
048    /**
049     * Gibt die Anzahl der Noten zurück.
050     * @return anzahl der Noten
051     */
052    public static int getCountOfGrades() {
053        return countGrades;
054    }
055
056    /**
057     * Gibt die Anzahl aller Bestandenen zurück.
058     * @return anzahl der Bestandnenen
059     */
060    public static int getCountOfPassed() {
061        return countPassed;
062    }
063
064    /**
065     * Füge eine Noten hinzu für die Berechnung.
066     * @param grade die Note als Double
067     */
068    public static void addGrade(double grade) {
069        countGrades++;
070        if (Grades.isPassed(grade)) {
071            countPassed++;
072            sumOfPassedGrades += grade;
073        }
074
075        bestGrade = Grades.isBetter(grade, bestGrade);
076        worstGrade = Grades.isWorse(grade, worstGrade);
077    }
078
079    /**
080     * Berechnet den Durchschnitt der Bestandenen und gibt diese zurück.
081     * @return durchschnitts Note der Bestandenen
082     */
083    public static double getAverageOfPassed() {
084        return sumOfPassedGrades / countPassed;
085    }
086
087    /**
088     * Berechnet die Durchfallquote und gibt diese zurück.
089     * @return durchfallquote
090     */
091    @SuppressWarnings("magicnumber")
092    public static double getFailureRate() {
093        return (double) (countGrades - countPassed) / (double) countGrades * 100;
094    }
095
096    /**
097     * gibt die beste Note zurück.
098     * @return beste Note als double (min. 1.0)
099     */
100    public static double getBestGrade() {
101        return bestGrade;
102    }
103
104    /**
105     * gibt die schlechteste Note zurück.
106     * @return schlechteste Note als double (max. 5.0)
107     */
108    public static double getWorstGrade() {
109        return worstGrade;
110    }
111
112    /** Gibt alle Werte in die Console an. */
113    public static void printInformation() {
114        System.out.printf("%nAnzahl beruecksichtigter Noten: %s%n", Klausurergebnis.getCountOfGrades());
115        System.out.printf("Anzahl Bestandene: %s%n", Klausurergebnis.getCountOfPassed());
116
117        if (Klausurergebnis.getCountOfPassed() > 0) {
118            System.out.printf("Beste Note: %s%n", Grades.toString(bestGrade));
119            System.out.printf("Schlechteste Note: %s%n", Grades.toString(worstGrade));
120            System.out.printf("Durchschnitt Bestandene: %s%n", Grades.toString(getAverageOfPassed()));
121            System.out.printf("Durchfallquote: %.1f%%%n", Klausurergebnis.getFailureRate());
122        }
123    }
124
125    private static final Scanner EINGABE = new Scanner(System.in);
126
127    /**
128     * main ist der Startpunkt des Programms.
129     * @param args wird nicht verwendet.
130     */
131    public static void main(String[] args) {
132        Locale.setDefault(Locale.GERMAN);
133
134        //--------------------------------------------------- Noten einlesen
135        System.out.println("Noten im Format Ganze,Zehntel "
136                           + "oder Ganze.Zehntel eingeben (Ende mit Strg-D):");
137
138        while (EINGABE.hasNext()) {
139            String strGrade = EINGABE.next();
140
141            //---------------------------------------------- Eingabe pruefen
142
143            /* (1) note pruefen ... */
144
145            if (!Grades.isCorrect(strGrade)) {
146                System.out.printf("Unzulaessige Note %s wird ignoriert!%n", strGrade);
147                continue;
148            }
149
150            //------------------------------------------------ Note erfassen
151
152            /* (2) Notensumme Bestandene, Anzahl Bestandene,
153                         Anzahl Durchgefallene sowie
154                         beste und schlechteste Note aktualisieren ...
155            */
156
157            double grade = Grades.toDouble(strGrade);
158
159            Klausurergebnis.addGrade(grade);
160
161        } // while
162
163        //------------------------------------------ Notenstatistik ausgeben
164
165        /* (3) Durchschnitt und Durchfallquote berechnen
166                     und dann die gesamte Statistik ausgeben ...
167        */
168
169        Klausurergebnis.printInformation();
170
171    } // main
172}
