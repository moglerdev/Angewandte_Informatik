003/**
004 * Klausurergebnis erstellt eine Notenstatistik.
005 * <p>
006 * Das Programm liest Noten als Strings ein und bestimmt die beste und
007 * die schlechteste Note, den Durchschnitt der Bestandenen sowie
008 * die Durchfallquote in Prozent.
009 * Das Programm ber&uuml;cksichtigt dabei nur die laut Noten.istZulaessig
010 * erlaubten Noten. Andere Noten werden unter Ausgabe einer Warnung ignoriert.
011 * Welche Noten besser und schlechter sind, welche als bestanden oder
012 * durchgefallen gelten und wie die String-Darstellung der Noten aussieht,
013 * wird mit Methoden der Klasse Noten bestimmt.
014 * </p>
015 * Das Programm gibt als Klausurergebnis folgende Werte aus:
016 * <ul>
017 * <li>die Anzahl der ber&uuml;cksichtigten Noten</li>
018 * <li>die Anzahl der Bestandenen</li>
019 * <li>die beste Note</li>
020 * <li>die schlechteste Note</li>
021 * <li>den Durchschnitt der Bestandenen</li>
022 * <li>die Durchfallquote</li>
023 * </ul>
024 *
025 * @author Christopher Mogler
026 * @version 02.12.2021
027 */
028@SuppressWarnings("magincnumber")
029public final class Grades {
030    /** Constructor. */
031    private Grades() {
032        super();
033    }
034
035    /** Beste mögliche Note. */
036    public static final double BESTE = 1.0;
037
038    /** Schlechteste mögliche Note. */
039    public static final double SCHLECHTESTE = 5.0;
040
041    /**
042     * Prüft ob die Eingabe als Note an der HTWG zulässig ist.
043     * @param grade die Note als String
044     * @return wenn zulässig <code>true</code>;
045     */
046    public static boolean isCorrect(String grade) {
047        boolean is4or5 = false;
048
049        if (grade.length() != 3) {
050            return false;
051        }
052
053        grade = grade.replace(',', '.');
054        if (grade.charAt(1) != '.') {
055            return false;
056        }
057
058        char pre = grade.charAt(0);
059        char sub = grade.charAt(2);
060
061        if (!(Character.isDigit(pre) || Character.isDigit(sub))) {
062            return false;
063        }
064
065        switch (pre) {
066        case '1':
067        case '2':
068        case '3':
069            break;
070        case '4':
071        case '5':
072            is4or5 = true;
073            break;
074        default:
075            return false;
076        }
077
078        switch (sub) {
079        case '0':
080            break;
081        case '3':
082        case '7':
083            if (is4or5) {
084                return false;
085            }
086            break;
087        default:
088            return false;
089        }
090
091        return true;
092    }
093
094    /**
095     * Konvertiert die Note als String zu einem Double-Typ um.
096     * @param grade true = 1.0 | 1.3 | 1.7 | 2.0 | 2.3 | 2.7 | 3.0 | 3.3 | 3.7 | 4.0 | 5.0
097     * @return Konvertiert die Eingabe der Note
098     */
099    public static double toDouble(String grade) {
100        if (!Grades.isCorrect(grade)) {
101            throw new IllegalArgumentException();
102        }
103
104        grade = grade.replace(',', '.');
105
106        return Double.parseDouble(grade);
107    }
108
109    /**
110     * Wandelt den Double-Typ in einen String mit nur einer Nachkommastelle.
111     * @param grade Note als double
112     * @return Note als String
113     */
114    public static String toString(double grade) {
115        if (grade > SCHLECHTESTE || grade < BESTE) {
116            throw new IllegalArgumentException();
117        }
118        return String.format("%.1f", grade);
119    }
120
121    /**
122     * Prüft ob die Note als Bestanden gilt.
123     * @param grade die Note
124     * @return gibt true zurück, wenn die die Note besser als 5.0 ist
125     */
126    public static boolean isPassed(double grade) {
127        return grade < Grades.SCHLECHTESTE;
128    }
129
130    /**
131     * Prüft ob die Note als Bestanden gilt.
132     * @param toTest die Note zu testenden
133     * @param curBetter die Note aktuell bessere
134     * @return gibt true zurück, wenn die die Note besser als 5.0 ist
135     */
136    public static double isBetter(double toTest, double curBetter) {
137        if (toTest < curBetter) {
138            return toTest;
139        }
140        return curBetter;
141    }
142
143    /**
144     * Prüft ob die Note als Bestanden gilt.
145     * @param toTest die Note zu testenden
146     * @param curWorse die Note aktuell schlechtere
147     * @return gibt true zurück, wenn die die Note besser als 5.0 ist
148     */
149    public static double isWorse(double toTest, double curWorse) {
150        if (toTest > curWorse) {
151            return toTest;
152        }
153        return curWorse;
154    }
155}
